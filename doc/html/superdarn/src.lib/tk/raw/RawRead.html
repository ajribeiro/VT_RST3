RawRead

Syntax
int RawRead(int fid,struct RadarParm *prm, struct RawData *raw);
Header
superdarn/rawread.h
Library
raw
Description

The RawRead function reads a rawacf data record from an open file.

The data is read from the file with the descriptor given by the argument fid. The data is decoded and used to populate the radar parameter block pointed to by the argument prm and the raw data structure pointer to by the argument raw.

Returns
Returns zero on success. On error, (-1) is returned.
Errors
On error, (-1) is returned.
Example

Source Code: RawRead.c

/* RawRead.c
   =========
   Author: R.J.Barnes */


#include <stdio.h>
#include <stdlib.h>

#include "rtypes.h"
#include "dmap.h"
#include "limit.h"
#include "rprm.h"
#include "rawdata.h"
#include "rawread.h"

int main(int argc,char *argv[]) {
 
  FILE *fp;
  int fd;

  struct RadarParm prm;
  struct RawData raw;

  fp=fopen(argv[1],"r");

  if (fp==NULL) {
    fprintf(stderr,"File not found.'n");
    exit(-1);
  }
  fd=fileno(fp);

  while(RawRead(fd,&prm,&raw) !=-1) {
    fprintf(stdout,"%.4d-%.2d-%.2d %.2d:%.2d:%.2d'n",
             prm.time.yr,prm.time.mo,prm.time.dy,
             prm.time.hr,prm.time.mt,prm.time.sc);
  }
 
  fclose(fp);


  return 0;
}
