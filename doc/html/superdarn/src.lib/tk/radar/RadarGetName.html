RadarGetName

Syntax
char *RadarGetName(struct RadarNetwork *ptr,int stid);
Header
superdarn/radar.h
Library
radar
Description

The RadarGetName function locates the station name for one of the radars.

The argument ptr is a pointer to a structure containing the information for the entire network. The radar identifier number for the radar is given by the argument stid.

Returns
Returns a pointer to a zero terminated text string that contains the radar name. On error, a NULL pointer is returned.
Errors
On error, a NULL pointer is returned.
Example

Source Code: RadarGetName.c

/* RadarGetName
   ============
   Author: R.J.Barnes */


#include <stdio.h>
#include <stdlib.h>

#include "rtypes.h"
#include "radar.h"

struct RadarNetwork *network; 

int main(int argc,char *argv[]) {
  char *envstr;
  FILE *fp;
  char *code=NULL;
  char *name=NULL;
  int st;

  envstr=getenv("SD_RADAR");
  if (envstr==NULL) {
    fprintf(stderr,"Environment variable 'SD_RADAR' must be defined.'n");
    exit(-1);
  }

  fp=fopen(envstr,"r");

  if (fp==NULL) {
    fprintf(stderr,"Could not locate radar information file.'n");
    exit(-1);
  }

  network=RadarLoad(fp);
  fclose(fp); 
  if (network==NULL) {
    fprintf(stderr,"Failed to read radar information.'n");
    exit(-1);
  }
  
  code=argv[argc-1];

  st=RadarGetID(network,code);
  name=RadarGetName(network,st);

  fprintf(stdout,"RadarGetName'n");
  fprintf(stdout,"%s=%s'n",code,name);

  return 0;
}
