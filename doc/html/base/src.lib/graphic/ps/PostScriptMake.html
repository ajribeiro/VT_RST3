PostScriptMake

Syntax
struct PostScript *PostScriptMake();
Header
base/rps.h
Library
rps
Description

The PostScriptMake function initializes an PostScript control structure.

Returns
Returns a pointer to the control structure on success. On error, a NULL pointer is returned.
Errors
On error, a NULL pointer is returned.

/* PostScriptMake
   ==============
   Author: R.J.Barnes
*/


#include <stdio.h>
#include <stdlib.h>
#include "rfbuffer.h"
#include "rps.h"

int stream(char *buf,int sze,void *data) {
  FILE *fp;
  fp=(FILE *) data;
  fwrite(buf,sze,1,stdout);
  return 0;
} 


int main(int argc,char *argv[]) {

  struct PostScript *ps=NULL;

  float wdt=400,hgt=400;
  float xoff=50,yoff=50;

  unsigned int fgcol=0xff000000;

  ps=PostScriptMake();
  PostScriptSetText(ps,stream,stdout);   
  PostScriptMakeDocument(ps,xoff,yoff,wdt,hgt,0);
  PostScriptMakePlot(ps);

  PostScriptRectangle(ps,NULL,25,25,wdt-50,hgt-50,
                 0,fgcol,0,NULL,NULL);

  PostScriptEndPlot(ps);
  PostScriptEndDocument(ps);
  return 0;
}
