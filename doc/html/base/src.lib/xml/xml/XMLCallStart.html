XMLCallStart

Syntax
int XMLCallStart(struct XMLdata *ptr, char *name,int atnum,char **atname,char **atval, char *buf,int sze);
Header
base/rxml.h
Library
rxml
Description

The XMLCallStart function forces a call to the user supplied function for interpreting the opening or start tags.

The data structure associated with the XML decoder is pointed to by the argument ptr.

The zero terminated string pointed to by the argument name gives the tag name.

The number of decoded attributes associated with the tag is given by the argument atnum. The name of each attribute is stored as a zero terminated string pointed to by the elements of the array atname. The associated value of each attribute is stored as a zero terminated string pointed to by the elements of the array atval.

The full text enclosed within the tag markers is stored in the buffer pointed to by the argument buf, and the size of this buffer is given by the argument sze.

Returns
Returns zero on success. On error, (-1) is returned.
Errors
On error (-1) is returned.
Example

Source Code: XMLCallStart.c

/* XMLCallStart.c
   ==============
   Author: R.J.Barnes
*/


#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include "rxml.h"

int start(char *name,int atnum,char **atname,char **atval,char *buf,
          int sze,void *data) {
  int a;
  FILE *fp=NULL;

  if (data !=NULL) fp=(FILE *) data;
  else fp=stderr;

  fprintf(fp,"'nStart Tag:%s'n",name);

  for (a=0;a<atnum;a++) { 
    fprintf(fp,"%s",atname[a]);
    if (atval[a] !=0) fprintf(fp,"=%s'n",atval[a]);
    else fprintf(fp,"'n");
  } 
  return 0;
}

int end(char *name,char *buf,
          int sze,void *data) {

  FILE *fp=NULL;
  if (data !=NULL) fp=(FILE *) data;
  else fp=stderr;
  fprintf(fp,"'nEnd Tag:%s'n",name);
  return 0;
}


int text(char *buf,
          int sze,void *data) {

  FILE *fp=NULL;
  if (data !=NULL) fp=(FILE *) data;
  else fp=stderr;
  fprintf(fp,"Body:");
  fwrite(buf,sze,1,fp);
  return 0;
}



int main(int argc,char *argv[]) {
 
  FILE *outp=stdout;
  char lbuf[256];

  char *atname[]={"width","height"};
  char *atval[]={"100","200"};

  struct XMLdata *xmldata;
 


  xmldata=XMLMake();

  XMLSetStart(xmldata,start,outp);
  XMLSetEnd(xmldata,end,outp);
  XMLSetText(xmldata,text,outp);

  sprintf(lbuf,"dummy contents.");

  XMLCallStart(xmldata,"table",2,atname,atval,"table",strlen("table"));

  XMLDecode(xmldata,lbuf,strlen(lbuf));

  XMLCallEnd(xmldata,"table","/table",strlen("/table"));

  XMLFree(xmldata);

  return 0;

}
